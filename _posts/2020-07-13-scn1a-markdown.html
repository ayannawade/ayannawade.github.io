# Import the counts and set up predictor objects #
{
  # read sample info
  sampleInfo <- read.table(samples, sep = "\t", header = T, as.is = T, stringsAsFactors = F)
  sampleInfo[is.na(sampleInfo)] <- 0
  
  # read count data
  countFull <- read.table(count, sep = "\t", header=T, as.is = T, stringsAsFactors = F)
  expData <- countFull[,c(2:ncol(countFull))]
  rownames(expData) <- countFull[,1]
  expData <- expData[,grep("Undetermined", colnames(expData), invert = T)]
  
  #remove samples 20, 23, 08, 14
  sampleInfo[which(sampleInfo[,"LibID"] == "DSRR00_20_S7"),"Genotype"] <- NA
  sampleInfo[which(sampleInfo[,"LibID"] == "DSRR00_23_S8"),"Genotype"] <- NA
  sampleInfo <- na.omit(sampleInfo)
  
  expData <- expData[,grep("DSRR00_20_S7", colnames(expData), invert = T)]
  expData <- expData[,grep("DSRR00_23_S8", colnames(expData), invert = T)]
  
  # covariates
  group <- sampleInfo[,"Genotype"]
  #sex <- sampleInfo[,"Sex"]
  sex_byRNA <- ifelse(expData["Xist",] > 100, "F","M")
  
  # rename columns 
  colnames(expData) <- paste(group, sex_byRNA, colnames(expData), sep="_")
}

#### DGE ####

min.cpm <- 1
min.lib <- 1

{
  # re-code covariates as factors #
  group <- as.factor(group)
  group <- relevel(group, ref = "WT")
  sex <- as.factor(ifelse(sex_byRNA == "M",1,2))

  # Set up edgeR expression object #
  y <- DGEList(counts = expData, group = group)
  keep <- rowSums(cpm(y) > min.cpm) >= min.lib
  y <- y[keep, , keep.lib.sizes=FALSE]
  
  # Perform simple exact test on genotype #
  y <- calcNormFactors(y)
  y <- estimateCommonDisp(y)
  y <- estimateTagwiseDisp(y)
  et <- exactTest(y)
  topTags(et)
  et$table[geneofInt,]
  et$table["Xist",]
  
  # Perform GLM for KO vs WT comparison
  #design <- model.matrix(~sex + group)
  design <- model.matrix(~group)
  y <- estimateGLMCommonDisp(y,design)
  y <- estimateGLMTrendedDisp(y,design)
  y <- estimateGLMTagwiseDisp(y,design)
  fit <- glmFit(y,design)
  
  lrt_3 <- glmLRT(fit, coef = 3)
  
  topTags(lrt_3, n=10)
  glm.output.lrt3 <- topTags(lrt_3, n=Inf)
  topTags(lrt_3)
  lrt_3$table[geneofInt,]
  lrt_3$table["Xist",]
  
  if (writeNewTables) {
    write.table(glm.output.lrt3$table, paste0("GlmOutput_KOvWt_", projName, "_", min.cpm, "cpm", projDate, ".txt"), sep="\t", col.names=T, row.names=T, quote=F)
  }
}

#### RPKM ####
  {
  if (runRPKM){
      
    # generate mm9 annotation data #
      
    txdb <- makeTxDbFromGFF(txdb_loc,format="gtf")
    exons.list.per.gene <- exonsBy(txdb,by="gene")
    exonic.gene.sizes <- unlist(lapply(exons.list.per.gene,function(x){sum(width(reduce(x)))}))
    gene_lengths <- vector(length=nrow(y))
      
    for (index in 1:nrow(y)) {
      gene_lengths[index] <- as.numeric(exonic.gene.sizes[rownames(y)[index]])
      }
      
    # generate pseudo counts and RPKM tables #
      
    pseudo_global <- y$pseudo.counts
    rownames(pseudo_global) <- rownames(y$counts)
    rpkmData <- rpkm(y, gene.length=gene_lengths, normalized.lib.sizes=TRUE, log=T, prior.count=.25)
    rpkmData_nolog <- rpkm(y, gene.length=gene_lengths, normalized.lib.sizes=TRUE, log=F, prior.count=.25)
      
    if (writeNewTables) {
      write.table(cbind(Gene=rownames(pseudo_global), pseudo_global), paste0("PseudoCounts_", projName, "_", min.cpm, "cpm", projDate,".txt"), sep="\t", col.names=T, row.names=F)
      write.table(cbind(Gene=rownames(rpkmData), rpkmData), paste0("RPKM_", projName, "_", min.cpm, "cpm", projDate, ".txt"), sep="\t", col.names=T, row.names=F)
      write.table(cbind(Gene=rownames(rpkmData_nolog), rpkmData_nolog), paste0("NoLogRPKM_", projName, "_", min.cpm, "cpm", projDate, ".txt"), sep="\t", col.names=T, row.names=F)
      }
      
    # generate cpm #
    scn1a.cpm <- cpm(y, prior.count=2, log=F)
    gene.lengths <- vector(length=nrow(y$counts))
    for (index in 1:length(gene.lengths)) {
      gene.lengths[index] <- as.numeric(exonic.gene.sizes[rownames(y$counts)[index]])
      }
   }
  }
